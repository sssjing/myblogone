                                                                                                    # 8.1 Vector

                                                                                                    • Vec<T>,叫做vector
                                                                                                      • 由标准库提供
                                                                                                      • 可存储多个值
                                                                                                      • 只能存储相同类型的数据
                                                                                                      • 值在内存中连续存放

                                                                                                    # 创建Vector

                                                                                                    • Vec::new 函数
                                                                                                      • let v: Vec<i32> = Vec::new();
                                                                                                    • 使用初始值创建Vec<T>（比较常用）,使用vec!宏
                                                                                                      • let v = vec![1,2,3];

                                                                                                    # 添加元素

                                                                                                    fn main(){
    let mut v = Vec::new();
    v.push(1);
    v.push(2);
}

                                                                                                    # 删除Vector

                                                                                                    • 与任何其他struct一样，当Vector离开作用域后
                                                                                                      • 它就会被清理
                                                                                                      • 它所有的元素也会被清理

                                                                                                    # 读取Vector的元素

                                                                                                    • 两种方式可以引用Vector里的值
                                                                                                      • 索引
                                                                                                      • get方法
                                                                                                    fn main() {
    // 用索引访问
    let v = vec![1, 2, 3, 4, 5];
    let a: &i32 = &v[2];
    println!("{}", a);
    
    // 用get访问
    match v.get(0) {
        Some(b) => println!("{}", b),
        None => println!("None")
    }
}

                                                                                                    # 索引 对比 get 处理访问越界

                                                                                                    • 索引：panic
                                                                                                    • get：返回None

                                                                                                    # 所有权和借用规则

                                                                                                    • 不能在同一作用域内同时拥有可变和不可变引用
                                                                                                    fn main(){
    let mut v = vec![1,2,3,4,5];
    let first = &v[0];
    v.push(6);
    println!("{}",first);
}

                                                                                                    报错

                                                                                                    以上代码会报错

                                                                                                    提示

                                                                                                    Vector在内存中的存放是连续的，在添加一个元素后，可能会对内存进行重新分配，原来的内存可能会重新释放

                                                                                                    let first = &v[0]还会指向原来的内存地址

                                                                                                    可能会导致bug

                                                                                                    # 遍历Vector中的值

                                                                                                    • for循环
                                                                                                    fn main() {
    let v = vec![100, 32, 57];
    for i in &v{//不可变引用
        println!("{}",i);
    }
}

                                                                                                    提示

                                                                                                    上方代码若直接使用for i in v则遍历完之后变量v则会失效

                                                                                                    fn main() {
    let mut v = vec![100, 32, 57];
    for i in &mut v{//可变引用
        *i += 50;
        println!("{}",i);
    }
}

                                                                                                    fn main() {
    let v = vec![100, 32, 57];
    for mut i in v{//可变引用
        i += 50;
        println!("{}",i);
    }
}

                                                                                                    # 使用枚举来存储多种数据类型

                                                                                                    • Enum的变体可以附加不同类型的数据
                                                                                                    • Enum的变体定义在同一个enum类型下
                                                                                                    enum SpreadsheetCell{
    Int(i32),
    Float(f64),
    Text(String),
}
fn main(){
    let row = vec![
        SpreadsheetCell::Int(3),
        SpreadsheetCell::Float(3.5),
        SpreadsheetCell::Text(String::from("blue")),
    ];
}

                                                                                                    # 8.3 String （上）

                                                                                                    # Rust开发正经常会被字符串困扰

                                                                                                    • Rust倾向于暴露可能的错误

                                                                                                    # 字符串是什么

                                                                                                    • Byte的集合

                                                                                                    • 一些方法

                                                                                                      • 能将byte解析为文本

                                                                                                    • Rust的核心语言层面，只有一个字符串类型：字符串切片str（或&str）

                                                                                                    • 字符串切片：对存储在其他地方、UTF-8编码的字符串的引用

                                                                                                      • 字符串字面值：存储在二进制文件中，也是字符串切片

                                                                                                    • String类型

                                                                                                      • 来自标准库而不是核心语言
                                                                                                      • 可增长、可修改、可拥有
                                                                                                      • UTF-8编码

                                                                                                    # 通常说的字符串是指

                                                                                                    • String和&str

                                                                                                    # 其它字符串类型

                                                                                                    • OsString,OsStr,CString,CStr
                                                                                                      • String对比Str后缀：拥有或借用的变体
                                                                                                      • 可存储不同编码的文本或内存中以不同的形式展现
                                                                                                    • Library crate（第三方库）针对春促字符串可提供更多的选项

                                                                                                    # 创建一个新的字符串（String）

                                                                                                    • 很多Vec<T>的操作都可用于String
                                                                                                    • String::new()函数
                                                                                                      • to_string()方法，可用于实现的Display trait的类型，包括字符串字面值
                                                                                                      • 例子
                                                                                                      fn main(){
  let data = "initial contents";
  let s = data.to_string();
  let s1 = "initial contents".to_string();
}

                                                                                                      提示

                                                                                                      s和s1这两种初始化方式等价

                                                                                                      • let s = String::from("initial contents");

                                                                                                    # 更新String

                                                                                                    • push_str()方法：把一个字符串切片附加到String，不会获得参数的所有权

                                                                                                        fn main() {
      let mut s = String::from("foo");
      s.push_str("bar");
      // println!("{s}");效果一样
      println!("{}",s);
  }

                                                                                                          fn main() {
      let mut s = String::from("foo");
      let s1 = String::from("bar");
      s.push_str(&s1);
      // println!("{s}");效果一样
      println!("{}",s);
  }

                                                                                                    • push()方法：把单个字符附加到String

                                                                                                    • +连接字符串

                                                                                                      let s1 = String::from("Hello, ");
let s2 = String::from("World");

let s3 = s1 + &s2;
println!("{s3}");

                                                                                                      提示

                                                                                                      使用+连接字符串使用了类似这个签名的方法

                                                                                                      fn add(self,s:&str) -> String {...}

                                                                                                      标准库中add方法使用了泛型

                                                                                                      只能把&str添加到String

                                                                                                      解引用强制换（deref coercion）

                                                                                                    • 使用format!连接多个字符串，不会取得任何参数的所有权

                                                                                                      fn main() {
    let s1 = String::from("Hello");
    let s2 = String::from("World");
    let s3 = String::from("China");
    //let s4 = s1 + "-" + &s2 + "-" + &s3;
    let s = format!("{s1}-{s2}-{s3}");
    println!("{s}")
}

                                                                                                    # 8.4 String(下)

                                                                                                    # 对String按索引的形式进行访问

                                                                                                    • 按索引语法访问String的某部分，会报错

                                                                                                    # 内部表示

                                                                                                    • String是对Vec<u8>的包装
                                                                                                      • len()方法
                                                                                                        fn main() {
      let s1 = String::from("Hello").len();
      //Unicode 标量值

      println!("{s1}");
  }

                                                                                                    # 字节、标量值、字形簇

                                                                                                    Bytes，Scalar Values，Grapheme Clusters

                                                                                                    • Rust有三种看待字符串的方式：
                                                                                                      • 字节
                                                                                                      • 标量值
                                                                                                      • 字形簇（最接近语言字母）
                                                                                                    • Rust不允许对String进行索引的原因
                                                                                                      • 索引操作应该消耗一个常量时间（O（1））
                                                                                                      • 而String无法保证：需要遍历所有内容，来确定有多少个合法字符

                                                                                                    # 切割String

                                                                                                    • 可以用[ ]和一个范围来创建字符串的切片
                                                                                                      • 必须谨慎使用
                                                                                                      • 如果切割时跨越了字符边界，程序就会panic！
                                                                                                    fn main(){
    let s = String::from("张羽丰");
    let s1 = &s[0..3];
    println!("{s1}");
    // 输出结果为 张
}

                                                                                                    # 遍历String的方法

                                                                                                    • 对于标量值chars()方法
                                                                                                    • 对于字节bytes()方法
                                                                                                    • 对于字形簇，标准库未提供

                                                                                                    # 总结

                                                                                                    • Rust选择将正确处理String数据作为所有Rust程序的默认行为
                                                                                                      • 程序员必须在处理UTF-8数据之前投入更多的精力
                                                                                                    • 可防止在开发后期处理涉及非ASCII字符的错误

                                                                                                    # 8.5 HashMap（上）

                                                                                                    # HashMap<K,V>

                                                                                                    • 键值对的形式存储数据，一个键（Key）对应一个值（Value）
                                                                                                    • Hash函数：决定如何在内存中存放K和V

                                                                                                    # 创建HashMap

                                                                                                    • 创建空HashMap new()函数
                                                                                                    • 添加数据 insert()方法
                                                                                                    use std::collections::HashMap;
fn main(){
    let mut scores: HashMap<String,i32> = HashMap::new();
    scores.insert(String::from("Blue"),10);
}
                                                                                                    • HashMap 不在prelude中
                                                                                                    • 标准库对其支持比较少，没有内置的宏来创建HashMap
                                                                                                    • 数据存储在heap上
                                                                                                    • HashMap是同构的
                                                                                                      • 所有的K必须是同一类型
                                                                                                      • 所有的V必须是同一类型

                                                                                                    # 使用collet方法创建HashMap

                                                                                                    只能在特定情况下使用

                                                                                                    • 在元素类型为Tuple的Vector上使用collect方法，可以组建一个HashMap
                                                                                                      • 要求Tuple有两个值：一个作为K、一个作为V
                                                                                                      • collect可以把数据组合成很多种集合类型，包括HashMap
                                                                                                        • 返回值需要显示指明类型
                                                                                                    use std::collections::HashMap;

fn main() {
    let teams: Vec<String> = vec![String::from("Blue"), String::from("Red")];
    let initial_scores: Vec<i32> = vec![10, 50];
    let scores: HashMap<String, i32> =
        teams.iter().zip(initial_scores.iter()).collect();
}

                                                                                                    # HashMap 和所有权

                                                                                                    • 对于实现了Copy trait的类型（例如i32），值会被复制到HashMap中
                                                                                                    • 队友拥有所有权的值（例如String），值会被移动，所有权会转移给HashMap
                                                                                                    use std::collections::HashMap;

fn main() {
    let field_name = String::from("yangjing");
    let field_value = String::from("stc");
    let mut map = HashMap::new();
    map.insert(field_name,field_value);
    println!("{:?}",map.get("yangjing"));
    // field_name field_value失效了
    //println!("{}:{}",field_name,field_value);
}
                                                                                                    • 如果将值的引用插入到HashMap中，值本身不会移动
                                                                                                      • map.insert(&field_name,&field_value);
                                                                                                      • 在HashMap有效的期间，被引用的值必须要保持有效

                                                                                                    # 访问HashMap中的值

                                                                                                    • get方法
                                                                                                      • 参数：K
                                                                                                      • 返回 Option<&V>
                                                                                                    use std::collections::HashMap;

fn main() {
    let mut scores = HashMap::new();

    scores.insert(String::from("zyf"), String::from("stc"));
    scores.insert(String::from("zyf1"), String::from("stc!"));

    let score = scores.get("zyf");

    match score {
        Some(s) => println!("{}", s),
        None => println!("team not exist"),
    };
}

                                                                                                    • for 循环
                                                                                                    for (k,v) in &scores{
    println!("{}:{}",k,v);
}

                                                                                                    # HashMap(下)

                                                                                                    # 更新HashMap

                                                                                                    • HashMap大小可变
                                                                                                    • 每个K同时只能对应一个V
                                                                                                    • 更新HashMap中的数据
                                                                                                      • K已经存在，对应一个V
                                                                                                        • 替换现有V
                                                                                                        • 保留现在的V，忽略新的V
                                                                                                        • 合并现有的V和新的V
                                                                                                      • K不存在
                                                                                                        • 添加新的

                                                                                                    # 替换现有的V

                                                                                                    • 如果向HashMap插入一对KV，然后再插入同样的K，但是不同的V，那么原来的V会被替换掉

                                                                                                    # 只在K不对应任何值的情况下，才插入V

                                                                                                    • entry 方法：检查指定的K是否对应一个V
                                                                                                      • 参数为K
                                                                                                      • 返回enum Entry：代表值是否存在
                                                                                                    use std::collections::HashMap;

fn main(){
    let mut scores = HashMap::new();

    scores.insert(String::from("Blue"),10);
    // scores.entry(String::from("Yellow")).or_insert(50);
    let e = scores.entry(String::from("Yellow"));
    println!("{:?}", e );
    e.or_insert(50);
    scores.entry(String::from("Blue")).or_insert(50);
    println!("{:?}",scores);
}
                                                                                                    • Entry的or_insert()方法
                                                                                                      • 返回：
                                                                                                        • 如果K存在，返回到对应的V的一个可变的引用
                                                                                                        • 如果K不存在，将方法参数作为K的新值插进去，返回到这个值的可变引用

                                                                                                    # 基于现有V更新V

                                                                                                    use std::collections::HashMap;

fn main() {
    let text = "hello world zyf zyf zyf world";
    let mut map = HashMap::new();

    for word in text.split_whitespace() {
        // entry(word)如果存在的话，就返回对应V可变引用
        let count = map.entry(word).or_insert(0);
        *count += 1;
    }

    println!("{:#?}", map);
}

                                                                                                    # Hash函数

                                                                                                    • 默认情况下，HashMap使用加密功能强大的Hash函数，可以抵抗拒绝服务（DoS）攻击
                                                                                                      • 不是可用的最快的Hash算法
                                                                                                      • 但具有更好的安全性
                                                                                                    • 可以指定不同的hasher来切换到另一个函数
                                                                                                      • hasher是实现BuildHasher trait的类型
